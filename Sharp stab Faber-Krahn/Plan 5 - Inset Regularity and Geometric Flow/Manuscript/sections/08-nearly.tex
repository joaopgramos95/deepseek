\section{Perturbative stability of nearly spherical sets}
\label{sec:nearly}

This section proves, in full and self-containedly, the perturbative
torsional stability of small-amplitude radial graphs consumed by the
reduction of \cref{sec:main}. The result is a Fuglede-type estimate
\emph{for the torsion functional}: a planar set whose boundary is a graph of
small amplitude over a circle, after the natural volume and barycentre
normalisations, has scale-invariant torsional deficit controlling the square
of its Fraenkel asymmetry. We invoke no selection principle, no spectral
black box, and no quantitative isoperimetric inequality.

The point requiring care is that the graph produced by \cref{sec:fold} has
\emph{small amplitude but no slope control}: its radial profile is only
Lipschitz, with a Lipschitz constant that may grow as $\dT\downarrow0$. The
classical nearly-spherical estimates (Fuglede~\cite{Fuglede1989} and their
torsional analogues) assume the boundary is a $C^1$- or $W^{2,\infty}$-small
graph; they do not apply here, and it is precisely to manufacture such
regularity that a selection principle is otherwise introduced. We therefore
give a proof valid under an $L^\infty$ amplitude bound alone. The mechanism
is an exact energy identity together with a \emph{slope-free} annular
Steklov lower bound that is diagonal in the boundary data; the coercive
output is at the $L^2$ (Fraenkel) scale, which is all that is needed and all
that survives at high frequency.

\subsection{Statement and standing normalisation}

For $\varphi\in\mathrm{Lip}(S^1)$ with $\|\varphi\|_{L^\infty}<1$, a radius
$\hat r_G>0$ and a centre $z\in\R^2$ we write
\begin{equation}\label{eq:Gdef}
   G=G(z,\hat r_G,\varphi):=\set{z+r\omega(\theta):\ \theta\in S^1,\
   0\le r<\hat r_G\,(1+\varphi(\theta))},\qquad \omega(\theta)=(\cos\theta,\sin\theta),
\end{equation}
the radial graph of profile $\varphi$. We recall that for a bounded open
$E\subset\R^2$ of finite measure with torsion function $u_E\in H^1_0(E)$
($-\Delta u_E=1$) and $T(E)=\int_E u_E$,
\begin{equation}\label{eq:dstarAdef}
   \dstar(E)=\frac{T(B_E)-T(E)}{T(B_E)},\quad T(B_E)=\frac{|E|^2}{8\pi},
   \qquad
   \AF(E)=\inf_{x_0}\frac{|E\triangle B(x_0,r_E)|}{|E|},\ r_E=\sqrt{|E|/\pi},
\end{equation}
are the scale-invariant torsional deficit and the Fraenkel asymmetry; both
are invariant under translations and dilations.

\begin{theorem}[Radial-graph torsion stability]\label{thm:graphG}
There are absolute constants $\eta_G\in(0,\tfrac1{13}]$ and $C_G<\infty$
\textup{(}one may take
\begin{equation}\label{eq:graphG-const}
   \eta_G=\tfrac1{80},\qquad C_G=3
\end{equation}
\textup{)} such that the following holds. Let $G=G(z,\hat r_G,\varphi)$ be a
radial graph as in \eqref{eq:Gdef} with $\varphi\in\mathrm{Lip}(S^1)$,
$\|\varphi\|_{L^\infty}\le\eta_G$, and barycentre at $z$. Then
\begin{equation}\label{eq:graphG}
   \AF(G)^2\ \le\ C_G\,\dstar(G).
\end{equation}
The statement is scale-invariant; $G$ need not have area $\pi$. It uses no
property of the original set $\Om$ and no selection.
\end{theorem}

\begin{remark}\label{rem:noslope}
The hypothesis is on the $L^\infty$ \emph{amplitude} of $\varphi$ alone: no
slope, perimeter, curvature, $C^1$ or $C^2$ bound is assumed. The amplitude
bound serves only to trap $\partial G$ in a thin annulus and to make the
volume and barycentre constraints remove the constant and first Fourier
modes quantitatively. This is exactly the regime delivered by
\cref{prop:realization}, where the amplitude is $O(\dT^{1/8})\to0$.
\end{remark}

Since $\dstar$ and $\AF$ are dilation- and translation-invariant, replacing
$G$ by $\hat r_G^{-1}(G-z)$ we assume throughout the proof
\begin{equation}\label{eq:norm}
   z=0,\qquad \hat r_G=1,\qquad |G|=\pi,
\end{equation}
so that $R:=1+\varphi$ satisfies the volume and barycentre constraints
\begin{equation}\label{eq:constraints}
   \int_{S^1}\bigl(R^2-1\bigr)\d\theta=0
   \quad\Bigl(\text{i.e. }\int_{S^1}(2\varphi+\varphi^2)\d\theta=0\Bigr),
   \qquad
   \int_{S^1}R^{3}\,\omega(\theta)\d\theta=0 .
\end{equation}
(The first is $|G|=\tfrac12\int R^2=\pi$; the second is
$\int_G x\,dx=\tfrac13\int R^3\omega=0$.) We write $u_0(x):=\tfrac14(1-|x|^2)$
for the disc torsion, $-\Delta u_0=1$, $u_0|_{\partial B_1}=0$,
$T(B_1)=\pi/8$. Expand $\varphi$ and the boundary datum
\begin{equation}\label{eq:hdef}
   h(\theta):=\frac{R(\theta)^2-1}{4}=\frac{2\varphi+\varphi^2}{4}
   =\frac\varphi2+\frac{\varphi^2}4
\end{equation}
in real Fourier series, with $P_k$ the projection onto the $k$-th mode
$\{\cos k\theta,\sin k\theta\}$ (for $k=0$ the constants) and
$P_{\ge2}=\sum_{k\ge2}P_k$. Recall $\|\psi\|_{L^2(S^1)}^2=\sum_{k\ge0}
\|P_k\psi\|_{L^2}^2$.

\subsection{An exact energy identity}

Write the energy functional and the (negative) torsion energy
\begin{equation}\label{eq:Jdef}
   J_E(v):=\tfrac12\int_E|\nabla v|^2-\int_E v\ \ (v\in H^1_0(E)),
   \qquad
   \mathcal E(E):=\min_{v\in H^1_0(E)}J_E(v)=J_E(u_E)=-\tfrac12 T(E),
\end{equation}
the last equality because $\int_E|\nabla u_E|^2=\int_E u_E=T(E)$. Thus
$\mathcal E(G)-\mathcal E(B_1)=\tfrac12\bigl(T(B_1)-T(G)\bigr)$, and on the
normalisation \eqref{eq:norm},
$\dstar(G)=\tfrac8\pi\bigl(\tfrac\pi8-T(G)\bigr)
=\tfrac{16}\pi\bigl(\mathcal E(G)-\mathcal E(B_1)\bigr)$.

\begin{lemma}[Exact graph identity]\label{lem:identity}
Let $z:=u_G-u_0\in H^1(G)$; then $z$ is harmonic in $G$ and $z=h$ on
$\partial G$ \textup{(}with $h$ from \eqref{eq:hdef}\textup{)}. Moreover
\begin{equation}\label{eq:identity8}
   \mathcal E(G)-\mathcal E(B_1)
   =\frac12\int_G|\nabla z|^2\d x
     -\frac1{32}\int_{S^1}\bigl(R^4-1\bigr)\d\theta .
\end{equation}
\end{lemma}

\begin{proof}
Both $u_G$ and $u_0$ solve $-\Delta(\cdot)=1$, so $z=u_G-u_0$ is harmonic in
$G$; on $\partial G=\{R(\theta)\theta\}$ we have $u_G=0$, hence
$z=-u_0(R\theta)=\tfrac14(R^2-1)=h$. Since $J_G$ is quadratic and
$u_G=u_0+z$,
\[
   \mathcal E(G)=J_G(u_G)=J_G(u_0)+\mathrm dJ_G(u_0)[z]+\tfrac12\int_G|\nabla z|^2 .
\]
The first variation is, using $-\Delta u_0=1$ and the divergence theorem on
the Lipschitz domain $G$,
\[
   \mathrm dJ_G(u_0)[z]=\int_G\nabla u_0\cdot\nabla z-\int_G z
   =\int_{\partial G}z\,\partial_\nu u_0\,d\HH^1 .
\]
On $\partial G$, $z=h=\tfrac14(R^2-1)$ and $\partial_\nu u_0=\nabla u_0\cdot
\nu=-\tfrac12\,x\cdot\nu$. For a radial graph $x=R(\theta)\omega(\theta)$ one
has the area element identity $(x\cdot\nu)\,d\HH^1=R^2\,d\theta$ \textup{(}the
outward normal is $\nu=(R\omega-R'\omega^\perp)/\sqrt{R^2+R'^2}$ and
$d\HH^1=\sqrt{R^2+R'^2}\,d\theta$, so $x\cdot\nu\,d\HH^1=R\,\omega\cdot
(R\omega)\,d\theta=R^2\,d\theta$\textup{)}. Hence
\[
   \mathrm dJ_G(u_0)[z]
   =\int_{S^1}\frac{R^2-1}{4}\cdot\Bigl(-\frac12\Bigr)R^2\d\theta
   =-\frac18\int_{S^1}(R^4-R^2)\d\theta .
\]
Next, with $u_0=\tfrac14(1-r^2)$, $|\nabla u_0|^2=r^2/4$, polar coordinates give
\[
   J_G(u_0)=\int_{S^1}\!\int_0^R\Bigl(\tfrac12\tfrac{r^2}4-\tfrac{1-r^2}4\Bigr)r\,dr\d\theta
   =\int_{S^1}\!\int_0^R\Bigl(\tfrac{3r^3}8-\tfrac r4\Bigr)dr\d\theta
   =\int_{S^1}\Bigl(\frac{3R^4}{32}-\frac{R^2}8\Bigr)\d\theta .
\]
Adding the two displays, the $R^2$ terms cancel:
\[
   \mathcal E(G)-\tfrac12\!\int_G|\nabla z|^2
   =\int_{S^1}\Bigl(\frac{3R^4}{32}-\frac{R^2}8\Bigr)\d\theta
   -\frac18\int_{S^1}(R^4-R^2)\d\theta
   =\int_{S^1}\Bigl(\frac{3R^4}{32}-\frac{R^4}8\Bigr)\d\theta
   =-\frac1{32}\int_{S^1}R^4\d\theta .
\]
Subtracting the same identity for $R\equiv1$ (where $z\equiv0$ and
$\mathcal E(B_1)=-\tfrac1{32}\!\int_{S^1}1=-\tfrac\pi{16}=-\tfrac12T(B_1)$,
consistently) yields \eqref{eq:identity8}.
\end{proof}

\subsection{A slope-free annular Steklov lower bound}

The next lemma is the heart of the matter: it bounds the Dirichlet energy of
\emph{any} $H^1$ function with prescribed outer trace $h$ from below by a
diagonal, $L^2$-coercive functional of $h$ --- with no reference to the
slope of $\partial G$.

Assume $1-\eta\le R\le1+\eta$ with $0<\eta\le\tfrac14$, and set
$\rho:=1-\eta$. Expand $h=\sum_{k\ge0}h_k$ into Fourier modes.

\begin{lemma}[Annular Steklov lower bound]\label{lem:annular}
Let $w\in H^1(G)$ have radial trace $w(R(\theta)\theta)=h(\theta)$ for a.e.\
$\theta$. Then
\begin{equation}\label{eq:annular}
   \int_G|\nabla w|^2\d x\ \ge\ b(\eta)\sum_{k\ge2}\|h_k\|_{L^2(S^1)}^2,
   \qquad
   b(\eta):=\frac{2\mu(\eta)}{2+\mu(\eta)},\quad \mu(\eta):=\frac{1-\eta}{2\eta},
\end{equation}
and $b(\eta)\uparrow2$ as $\eta\downarrow0$; in particular $b(\eta)\ge\tfrac32$
whenever $\eta\le\tfrac1{13}$.
\end{lemma}

\begin{proof}
Let $f(\theta):=w(\rho\,\theta)$ be the trace of $w$ on the inner circle
$\partial B_\rho\subset G$ (it lies in $G$ since $\rho=1-\eta\le R$). Split
$\int_G|\nabla w|^2=\int_{B_\rho}|\nabla w|^2+\int_{G\setminus B_\rho}
|\nabla w|^2$.

\emph{Inner ball.} Among all $H^1(B_\rho)$ functions with trace $f$ on
$\partial B_\rho$ the Dirichlet energy is minimised by the harmonic
extension, so with $f=\sum_k f_k$,
\begin{equation}\label{eq:inner}
   \int_{B_\rho}|\nabla w|^2\ \ge\ \sum_{k\ge1}k\,\|f_k\|_{L^2(S^1)}^2
\end{equation}
(in the plane the harmonic extension of the mode $f_k$ is $(r/\rho)^k$ times
it, with Dirichlet energy $k\|f_k\|_{L^2}^2$, independent of $\rho$).

\emph{Annular shell.} For a.e.\ $\theta$, along the radial segment
$r\in[\rho,R(\theta)]$ Cauchy--Schwarz gives
\[
   |h(\theta)-f(\theta)|^2
   =\Bigl(\int_\rho^{R(\theta)}\!\partial_r w\,dr\Bigr)^2
   \le\Bigl(\int_\rho^{R(\theta)}\!|\partial_r w|^2\,r\,dr\Bigr)
      \Bigl(\int_\rho^{R(\theta)}\!\frac{dr}{r}\Bigr).
\]
Since $R\le1+\eta$, $\rho=1-\eta$ and $\log\tfrac{1+t}{1-t}\le\tfrac{2t}{1-t}$
for $t\in[0,\tfrac14]$,
$\int_\rho^{R}\tfrac{dr}r=\log\tfrac{R}\rho\le\log\tfrac{1+\eta}{1-\eta}\le
\tfrac{2\eta}{1-\eta}=\mu(\eta)^{-1}$. Hence
$\int_\rho^{R(\theta)}|\partial_r w|^2 r\,dr\ge\mu(\eta)|h(\theta)-f(\theta)|^2$,
and integrating in $\theta$,
\begin{equation}\label{eq:shell}
   \int_{G\setminus B_\rho}|\nabla w|^2\ \ge\ \mu(\eta)\,\|h-f\|_{L^2(S^1)}^2
   =\mu(\eta)\sum_{k\ge0}\|h_k-f_k\|_{L^2}^2 .
\end{equation}

\emph{Mode-by-mode minimisation.} Adding \eqref{eq:inner} and
\eqref{eq:shell} (and dropping the $k=0$ shell term) gives the lower bound
$\sum_{k\ge1}\bigl(k\|f_k\|^2+\mu\|h_k-f_k\|^2\bigr)$. Minimising
$k|f_k|^2+\mu|h_k-f_k|^2$ over $f_k$ yields $\frac{k\mu}{k+\mu}|h_k|^2$. The
coefficient $\frac{k\mu}{k+\mu}$ is increasing in $k$, so for $k\ge2$ it is
at least its value at $k=2$, namely $\frac{2\mu}{2+\mu}=b(\eta)$. This is
\eqref{eq:annular}. Finally $b(\eta)\ge\tfrac32\iff\mu\ge6\iff
\frac{1-\eta}{2\eta}\ge6\iff\eta\le\tfrac1{13}$.
\end{proof}

\subsection{Removing the neutral modes}

\begin{lemma}[Neutral modes]\label{lem:modes}
There is an absolute $C_0$ such that if $\|\varphi\|_{L^\infty}\le\tfrac1{13}$
and \eqref{eq:constraints} hold, then
\begin{equation}\label{eq:neutral}
   \|P_0\varphi\|_{L^2}+\|P_1\varphi\|_{L^2}\ \le\ C_0\,\|\varphi\|_{L^2}^2 .
\end{equation}
Consequently there is an absolute $\eps_1>0$ such that
$\|\varphi\|_{L^2}\le\eps_1$ implies
$\|P_{\ge2}\varphi\|_{L^2}^2\ge\tfrac{15}{16}\|\varphi\|_{L^2}^2$.
\end{lemma}

\begin{proof}
The volume constraint $\int_{S^1}(2\varphi+\varphi^2)=0$ gives
$2\int\varphi=-\int\varphi^2$, so the mean obeys
$|\overline\varphi|=\tfrac1{2\pi}|\int\varphi|=\tfrac1{4\pi}\int\varphi^2$ and
$\|P_0\varphi\|_{L^2}=\sqrt{2\pi}\,|\overline\varphi|
=\tfrac1{2\sqrt{2\pi}}\|\varphi\|_{L^2}^2$. The barycentre constraint with
$R^3=1+3\varphi+3\varphi^2+\varphi^3$ and $\int_{S^1}\omega=0$ gives
$3\int\varphi\,\omega=-\int(3\varphi^2+\varphi^3)\omega$, whence, using
$\|P_1\varphi\|_{L^2}=\tfrac1{\sqrt\pi}|\int\varphi\,\omega|$ and
$\|\varphi\|_{L^\infty}\le\tfrac1{13}$,
\[
   \|P_1\varphi\|_{L^2}\le\frac1{3\sqrt\pi}\int(3\varphi^2+|\varphi|^3)
   \le\frac{1+\tfrac1{39}}{\sqrt\pi}\,\|\varphi\|_{L^2}^2 .
\]
This is \eqref{eq:neutral} with $C_0=\tfrac1{2\sqrt{2\pi}}+\tfrac{1+1/39}{\sqrt\pi}$.
Then $\|P_0\varphi\|^2+\|P_1\varphi\|^2\le2C_0^2\|\varphi\|_{L^2}^4\le
\tfrac1{16}\|\varphi\|_{L^2}^2$ once $\|\varphi\|_{L^2}^2\le\eps_1^2:=
1/(32C_0^2)$, and the last claim follows.
\end{proof}

\begin{lemma}[Boundary datum and moment]\label{lem:taylor}
If $\|\varphi\|_{L^\infty}\le\eta\le\tfrac1{13}$ and \eqref{eq:constraints}
holds, then
\begin{align}
   \int_{S^1}\bigl(R^4-1\bigr)\d\theta
      &\le\ \bigl(4+4\eta+\eta^2\bigr)\,\|\varphi\|_{L^2}^2 ,
      \label{eq:moment}\\
   \|P_{\ge2}h\|_{L^2}^2
      &\ge\ \tfrac14\,\|P_{\ge2}\varphi\|_{L^2}^2
            -\tfrac14\,\eta\,\|\varphi\|_{L^2}^2 .
      \label{eq:hlow}
\end{align}
\end{lemma}

\begin{proof}
Expand $R^4-1=4\varphi+6\varphi^2+4\varphi^3+\varphi^4$. By
\eqref{eq:constraints}, $4\int\varphi=-2\int\varphi^2$, so
\[
   \int_{S^1}(R^4-1)=4\int\varphi^2+\int(4\varphi^3+\varphi^4)
   \le\bigl(4+4\eta+\eta^2\bigr)\|\varphi\|_{L^2}^2,
\]
using $\int|4\varphi^3+\varphi^4|\le(4\eta+\eta^2)\int\varphi^2$; this is
\eqref{eq:moment}. For \eqref{eq:hlow}, from \eqref{eq:hdef},
$P_{\ge2}h=\tfrac12P_{\ge2}\varphi+\tfrac14P_{\ge2}(\varphi^2)$ with
$\|P_{\ge2}(\varphi^2)\|_{L^2}\le\|\varphi^2\|_{L^2}\le\|\varphi\|_{L^\infty}
\|\varphi\|_{L^2}\le\eta\|\varphi\|_{L^2}$. Hence, by the triangle inequality,
$\|P_{\ge2}h\|_{L^2}\ge\tfrac12\|P_{\ge2}\varphi\|_{L^2}-\tfrac14\eta
\|\varphi\|_{L^2}$, and both terms are nonnegative; squaring and discarding
the nonnegative square term,
\[
   \|P_{\ge2}h\|_{L^2}^2\ge\tfrac14\|P_{\ge2}\varphi\|_{L^2}^2
   -2\cdot\tfrac12\cdot\tfrac14\eta\|P_{\ge2}\varphi\|_{L^2}\|\varphi\|_{L^2}
   \ge\tfrac14\|P_{\ge2}\varphi\|_{L^2}^2-\tfrac14\eta\|\varphi\|_{L^2}^2,
\]
using $\|P_{\ge2}\varphi\|_{L^2}\le\|\varphi\|_{L^2}$ in the last step.
\end{proof}

\subsection{Proof of \texorpdfstring{\cref{thm:graphG}}{the theorem}}

By \cref{prop:realization} the graphs entering the reduction have amplitude
$\|\varphi\|_{L^\infty}\le\eta\le C\dT^{1/8}$, so for $\dT$ small both
$\|\varphi\|_{L^\infty}\le\eta_G=\tfrac1{80}$ and (since
$\|\varphi\|_{L^2}\le\sqrt{2\pi}\,\|\varphi\|_{L^\infty}\le\sqrt{2\pi}/80
\le\eps_1$, with $\eps_1$ from \cref{lem:modes}) the hypotheses below hold.
We prove \eqref{eq:graphG} for every $\varphi$ with
\begin{equation}\label{eq:thresh}
   \|\varphi\|_{L^\infty}\le\eta_G=\tfrac1{80},
\end{equation}
which already forces $\|\varphi\|_{L^2}\le\sqrt{2\pi}/80<\eps_1$, so
\cref{lem:modes,lem:taylor} apply with $\eta=\tfrac1{80}$.

By \cref{lem:annular} applied to $w=z$ (legitimate: $z$ is harmonic with
outer trace $h$, and $1-\eta_G\le R\le1+\eta_G$ with $\eta_G\le\tfrac1{13}$,
so $b(\eta_G)\ge\tfrac32$), then \eqref{eq:hlow} and the
$\|P_{\ge2}\varphi\|^2\ge\tfrac{15}{16}\|\varphi\|^2$ of \cref{lem:modes},
\begin{equation}\label{eq:zlow}
   \int_G|\nabla z|^2
   \ \ge\ \tfrac32\,\|P_{\ge2}h\|_{L^2}^2
   \ \ge\ \tfrac32\Bigl(\tfrac14\cdot\tfrac{15}{16}-\tfrac14\eta_G\Bigr)
          \|\varphi\|_{L^2}^2
   \ =\ \Bigl(\tfrac{45}{128}-\tfrac38\eta_G\Bigr)\|\varphi\|_{L^2}^2 .
\end{equation}
Combining \eqref{eq:zlow} and the moment bound \eqref{eq:moment} in the exact
identity \eqref{eq:identity8},
\begin{equation}\label{eq:deffinal}
\begin{aligned}
   \mathcal E(G)-\mathcal E(B_1)
   &\ge\ \frac12\Bigl(\tfrac{45}{128}-\tfrac38\eta_G\Bigr)\|\varphi\|_{L^2}^2
        -\frac1{32}\bigl(4+4\eta_G+\eta_G^2\bigr)\|\varphi\|_{L^2}^2\\
   &=\ \Bigl(\tfrac{45}{256}-\tfrac18-\tfrac5{16}\eta_G-\tfrac1{32}\eta_G^2\Bigr)
        \|\varphi\|_{L^2}^2
   \ =\ \Bigl(\tfrac{13}{256}-\tfrac5{16}\eta_G-\tfrac1{32}\eta_G^2\Bigr)
        \|\varphi\|_{L^2}^2 .
\end{aligned}
\end{equation}
With $\eta_G=\tfrac1{80}$ the bracket equals
$\tfrac{13}{256}-\tfrac5{1280}-\tfrac1{204800}=0.04687\ge0.046=:c_0$, so
\begin{equation}\label{eq:Edown}
   \mathcal E(G)-\mathcal E(B_1)\ \ge\ c_0\,\|\varphi\|_{L^2}^2,
   \qquad c_0=0.046,
\end{equation}
and therefore, by $\dstar(G)=\tfrac{16}\pi(\mathcal E(G)-\mathcal E(B_1))$,
\begin{equation}\label{eq:dstarlow}
   \dstar(G)\ \ge\ \frac{16 c_0}\pi\,\|\varphi\|_{L^2}^2
   \ =\ \frac{0.736}\pi\,\|\varphi\|_{L^2}^2
   \ \ge\ 0.234\,\|\varphi\|_{L^2}^2 .
\end{equation}

Finally we bound the asymmetry. Comparing $G$ with $B_1$,
\[
   |G\triangle B_1|=\int_{S^1}\Bigl|\int_0^{R}r\,dr-\int_0^1 r\,dr\Bigr|\d\theta
   =\frac12\int_{S^1}|R^2-1|\d\theta
   =\frac12\int_{S^1}|2\varphi+\varphi^2|\d\theta
   \le\bigl(1+\tfrac{\eta_G}2\bigr)\!\int_{S^1}\!|\varphi|\d\theta,
\]
so by Cauchy--Schwarz $|G\triangle B_1|\le(1+\tfrac{\eta_G}2)\sqrt{2\pi}\,
\|\varphi\|_{L^2}$, and since $\AF(G)\le|G\triangle B_1|/\pi$,
\begin{equation}\label{eq:Aupper}
   \AF(G)^2\ \le\ \frac{2(1+\eta_G/2)^2}\pi\,\|\varphi\|_{L^2}^2
   \ \le\ \frac{2(1+1/160)^2}\pi\,\|\varphi\|_{L^2}^2\ \le\ 0.645\,\|\varphi\|_{L^2}^2 .
\end{equation}
Combining \eqref{eq:Aupper} with \eqref{eq:dstarlow},
\[
   \AF(G)^2\ \le\ 0.645\,\|\varphi\|_{L^2}^2
   \ \le\ \frac{0.645}{0.234}\,\dstar(G)\ =\ 2.76\,\dstar(G)\ \le\ 3\,\dstar(G),
\]
which is \eqref{eq:graphG} with $C_G=3$.\qed

\begin{remark}[Sharpness and what is used]\label{rem:sharp}
The exponent $2$ in \eqref{eq:graphG} is sharp: for $\varphi=a\cos2\theta$
(a small ellipse) one has $\AF(G)^2\sim c_1a^2$ and $\dstar(G)\sim c_2a^2$ as
$a\to0$ with $c_1,c_2>0$, so $\AF(G)^2/\dstar(G)$ tends to a finite positive
limit and no exponent larger than $2$ can hold. The proof uses only: the
exact energy identity \cref{lem:identity} (integration by parts and the
radial-graph area element); the Dirichlet principle and a one-dimensional
Cauchy--Schwarz along radii (\cref{lem:annular}); the harmonicity of the
disc modes $r^k e^{\mathrm i k\theta}$; Parseval; and the volume/barycentre
constraints. There is \emph{no} quantitative isoperimetric inequality,
\emph{no} selection or symmetrisation, \emph{no} second shape derivative,
and crucially \emph{no} slope or curvature control on $\partial G$ --- only
the $L^\infty$ amplitude bound. The coercive output is at the $L^2$
(Fraenkel) scale; the stronger $\dot H^{1/2}$ bound is false at this level of
generality (high-frequency ripples) but is not needed. The constants
\eqref{eq:graphG-const} are not optimised; the threshold $\eta\le1/13$ for
$b(\eta)\ge3/2$ in \cref{lem:annular} is the only place the amplitude enters
the coercivity, and any smaller absolute $\eta_G$ is admissible, which is all
\cref{sec:main} requires since the inset amplitude is $O(\dT^{1/8})\to0$.
\end{remark}
