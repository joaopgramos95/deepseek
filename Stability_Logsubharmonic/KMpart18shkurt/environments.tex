%%%%%%%%%%%%%%%%
% ENVIRONMENTS %
%%%%%%%%%%%%%%%%

\newcommand{\theoremname}{Theorem}
\newcommand{\lemmaname}{Lemma}
\newcommand{\propositionname}{Proposition}
\newcommand{\definitionname}{Definition}
\newcommand{\corollaryname}{Corollary}
\newcommand{\claimname}{Claim}
\newcommand{\exercisename}{Exercise}
\newcommand{\examplename}{Example}
\newcommand{\remarkname}{Remark}
\newcommand{\commentsname}{Comments}
\newcommand{\notename}{Note}
\newcommand{\notationname}{Notation}
\newcommand{\conjecturename}{Conjecture}

\newcounter{allcounter}[section]
\renewcommand{\theallcounter}{\thesection.\arabic{allcounter}}

% Theorem-like environments
\theoremstyle{plain}
\newtheorem*{thm}{\theoremname}
\newtheorem{numthm}[allcounter]{\theoremname}
\newtheorem{numlemma}[allcounter]{\lemmaname}
\newtheorem{numprop}[allcounter]{\propositionname}
\newtheorem{numcor}[allcounter]{\corollaryname}
\newtheorem{numrem}[allcounter]{\remarkname}
\newtheorem{numconj}[allcounter]{\conjecturename}

% Definition-like environments
\theoremstyle{definition}
\newtheorem*{defn}{\definitionname}
\newtheorem{numdefn}[allcounter]{\definitionname}
\newtheorem*{examp}{\examplename}
\newtheorem{numexamp}[allcounter]{\examplename}

% Remark-like environments
\theoremstyle{remark}
\newtheorem*{obs}{\remarkname}
\newtheorem*{comment}{\commentsname}
\newtheorem*{note}{\notename}
\newtheorem*{notation}{\notationname}
\newtheorem{numremark}[allcounter]{\remarkname}
