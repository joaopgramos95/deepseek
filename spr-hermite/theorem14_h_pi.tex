% Frank--Nicola--Tilli, Theorem 14 (stability of log-Sobolev for the Fock space)
% specialized to h = \pi.
% With h = \pi, the probability measure becomes
%   d\mu_\pi(z) = \pi^{-1} e^{-|z|^2}\, dA(z),
% the prefactor h/\pi equals 1, and the optimizers reduce to
%   F(z) = c\, e^{\beta z - |\beta|^2/2}, \quad \beta \in \mathbb{C},\ |c|=1.

\begin{theorem}[Frank--Nicola--Tilli, {\cite[Thm.~14]{FrankNicolaTilli2025}}, $h=\pi$]
There exists a constant $c_\ast > 0$ such that, for every
$F \in \mathcal{F}^2_\pi$ with $\|F\|_{\mathcal{F}^2_\pi} = 1$,
\begin{equation}
  \int_{\mathbb{C}} |\partial_z F(z)|^2 \, d\mu_\pi(z)
  \;\ge\;
  \int_{\mathbb{C}} |F(z)|^2 \ln |F(z)|^2 \, d\mu_\pi(z)
  \;+\;
  c_\ast \!\!\inf_{\substack{\beta \in \mathbb{C} \\ |c|=1}}\!
  \bigl\| F - c\, e^{\beta z - |\beta|^2/2} \bigr\|_{\mathcal{F}^2_\pi}^{2},
\end{equation}
where $d\mu_\pi(z) = \pi^{-1} e^{-|z|^2}\, dA(z)$ and $\mathcal{F}^2_\pi$
denotes the Fock space of entire functions square-integrable with respect
to $\mu_\pi$.
\end{theorem}
